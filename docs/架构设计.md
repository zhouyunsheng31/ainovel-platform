## 二、数据库设计

### 2.1 数据库选型

| 数据类型 | 存储方案 | 理由 |
|---------|---------|------|
| 书籍元数据 | SQLite | 结构化数据，查询简单，轻量级 |
| 文件存储 | 本地文件系统 | 小B端，单机部署，无需对象存储 |
| 章节与纲数据 | SQLite | 关系型数据，支持关联查询 |
| 进度状态 | 内存 + SQLite | 实时进度用内存，持久化用SQLite |
| WebSocket连接 | 内存 | 临时状态，无需持久化 |

### 2.3 表结构详细设计

#### 2.3.1 books 表

```sql
CREATE TABLE books (
    book_id TEXT PRIMARY KEY,           -- UUID
    title TEXT NOT NULL,                -- 书名（从文件提取或用户指定）
    original_name TEXT NOT NULL,        -- 原始文件名
    file_type TEXT NOT NULL,            -- TXT/EPUB/DOC/DOCX/PDF
    file_path TEXT NOT NULL,            -- 文件存储路径
    file_size INTEGER NOT NULL,         -- 文件大小(bytes)
    encoding TEXT,                      -- 文本编码(UTF-8/GBK/GB2312)
    total_chapters INTEGER DEFAULT 0,   -- 总章节数
    status TEXT DEFAULT 'IDLE',         -- IDLE/UPLOADING/PREPROCESSING/EXTRACTING/GENERATING/OUTLINING/WORLDING/COMPLETED/ERROR
    created_at TEXT NOT NULL,           -- ISO8601格式时间
    updated_at TEXT NOT NULL
);

CREATE INDEX idx_books_status ON books(status);
CREATE INDEX idx_books_created_at ON books(created_at);
```

#### 2.3.2 processing_tasks 表

```sql
CREATE TABLE processing_tasks (
    task_id TEXT PRIMARY KEY,           -- UUID
    book_id TEXT NOT NULL,              -- 关联书籍
    current_stage TEXT DEFAULT 'UPLOADING', -- FILE_UPLOAD/TEXT_PREPROCESS/CHAPTER_OUTLINE/COARSE_OUTLINE/MAIN_OUTLINE/WORLD_OUTLINE
    stage_progress DEFAULT '{}',        -- JSON: {"UPLOADING": 100, "EXTRACTING": 45, ...}
    total_chapters INTEGER DEFAULT 0,
    completed_chapters INTEGER DEFAULT 0,
    status TEXT DEFAULT 'PENDING',      -- PENDING/RUNNING/COMPLETED/FAILED
    start_time TEXT,                    -- ISO8601
    end_time TEXT,                      -- ISO8601
    created_at TEXT NOT NULL,
    FOREIGN KEY (book_id) REFERENCES books(book_id)
);

CREATE INDEX idx_tasks_book_id ON processing_tasks(book_id);

class MetricsCollector:
    def __init__(self):
        self.counters = defaultdict(int)
        self.histograms = defaultdict(list)
    
    def increment(self, name: str, value: int = 1, labels: dict = None):
        key = f"{name}_{json.dumps(labels or {})}"
        self.counters[key] += value
    
    def observe(self, name: str, value: float, labels: dict = None):
        key = f"{name}_{json.dumps(labels or {})}"
        self.histograms[key].append(value)
    
    def get_metrics(self) -> dict:
        return {
            "counters": dict(self.counters),
            "histograms": {k: {"count": len(v), "avg": sum(v)/len(v)} 
                          for k, v in self.histograms.items()}
        }

#### 4.3.3 埋点实现示例

**前端埋点：**
```javascript
// 简单埋点收集器
class Tracker {
    constructor() {
        this.events = [];
        this.sessionId = this.generateSessionId();
    }

    track(eventName, data = {}) {
        const event = {
            event: eventName,
            timestamp: new Date().toISOString(),
            sessionId: this.sessionId,
            ...data
        };
        this.events.push(event);
        this.sendEvents();
    }

    async sendEvents() {
        if (this.events.length >= 10) {
            await fetch('/api/track', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ events: this.events })
            });
            this.events = [];
        }
    }
}

// 使用示例
tracker.track('file_upload_start', {
    file_type: 'txt',
    file_size: 1024000
});
```

**后端埋点：**
```python
# 埋点装饰器
import functools
import time
from datetime import datetime

def track_api(func):
    @functools.wraps(func)
    async def wrapper(*args, **kwargs):
        start_time = time.time()
        try:
            result = await func(*args, **kwargs)
            status = "success"
            return result
        except Exception as e:
            status = "error"
            raise e
        finally:
            duration = time.time() - start_time
            track_event("api_request", {
                "method": func.__name__,
                "status": status,
                "duration_ms": duration * 1000,
                "timestamp": datetime.utcnow().isoformat()
            })
    return wrapper

# 使用示例
@track_api
async def upload_file(file: UploadFile):
    # ... 处理逻辑
    pass
```
